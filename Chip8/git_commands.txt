git pull - получить коммиты для текущей ветки с сервера
git status - текущее состояние локальных файлов
git add [файлы] - добавить файл к репозиторию или к коммиту
git restore [файлы] - отменить локальные изменения отдельных файлов так, чтобы было как на сервере в текущей ветке
git reset [ветка] --hard - откатить все локальные изменения так, чтобы было как в указанной ветке
git commit -m"текст" - собрать коммит из добавленных файлов и снабдить его комментарием
git push origin - отправить коммиты для текущей ветки на сервер
git log - список коммитов в текущей ветке
git checkout -b [имя новой ветки] - создать новую ветку с корнем в текущем коммите
git checkout [имя ветки] - переключиться на существующую ветку
